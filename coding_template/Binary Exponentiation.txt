const int mod = 1e9+7;
int binexp(int a, int b) {
    int res = 1;
    while (b > 0){if (b & 1){res = res * a;}a = a * a;b >>= 1;}
    return res;
}
int binpow( int a,int b, int mod) {
    a %=mod;
    int res = 1;
    while (b > 0){if (b & 1){res = res * a % mod;}a = a * a % mod;b >>= 1;}
    return res;
}
